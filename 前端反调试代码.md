from playwright.sync_api import sync_playwright
import pandas as pd

with sync_playwright() as p:
    browser = p.chromium.launch(headless=False)
    context = browser.new_context(ignore_https_errors=True)
    
    # ✅ 只覆盖webdriver检测，不动其他任何东西
    context.add_init_script("""
        Object.defineProperty(navigator, 'webdriver', {get: () => undefined});
    """)
    
    page = context.new_page()
    
    # ✅ 监听所有网络请求，找到数据接口
    requests_log = []
    def on_request(request):
        if request.resource_type in ["xhr", "fetch"]:
            requests_log.append(request.url)
            print("📡 发现请求:", request.url)
    
    page.on("request", on_request)
    
    page.goto("http://www.aqistudy.cn/historydata/monthdata.php?city=北京")
    page.wait_for_timeout(15000)  # 等15秒
    
    print("\n=== 所有XHR/Fetch请求 ===")
    for r in requests_log:
        print(r)
    
    browser.close()

                let line = [];
                cells.forEach(cell => line.push(cell.innerText.trim()));
                if(line.length > 0) result.push(line);
            });

        print("✅ 保存成功！")

from playwright.sync_api import sync_playwright
import pandas as pd

with sync_playwright() as p:
    browser = p.chromium.launch(headless=False)
    context = browser.new_context(ignore_https_errors=True)
    context.add_init_script("""
        Object.defineProperty(navigator, 'webdriver', {get: () => undefined});
    """)
    
    page = context.new_page()
    page.goto("http://www.aqistudy.cn/historydata/monthdata.php?city=北京")
    page.wait_for_timeout(15000)
    
    # 同时读取表头和数据，保持列对应关系
    result = page.evaluate("""
        () => {
            let table = document.querySelector('table');
            let headers = Array.from(table.querySelectorAll('th')).map(e => e.innerText.trim());
            let rows = [];
            table.querySelectorAll('tbody tr').forEach(row => {
                let cells = Array.from(row.querySelectorAll('td')).map(e => e.innerText.trim());
                if(cells.length > 0) rows.push(cells);
            });
            return {headers, rows};
        }
    """)
    
    headers = result['headers']
    data = result['rows']
    
    print(f"表头数量: {len(headers)}, 数据列数: {len(data[0])}")
    print("表头:", headers)
    print("第一行数据:", data[0])
    
    # ✅ 找到月份列索引

from playwright.sync_api import sync_playwright
import pandas as pd

with sync_playwright() as p:
    browser = p.chromium.launch(headless=False)
    context = browser.new_context(ignore_https_errors=True)
    
    context.add_init_script("""
        Object.defineProperty(navigator, 'webdriver', {get: () => undefined});
        window.__intercepted_data__ = null;
        
        const _parse = JSON.parse;
        JSON.parse = function(text) {
            try {
                const result = _parse(text);
                if (result && result.success && result.result && result.result.data) {
                    window.__intercepted_data__ = result.result.data;
                }
                return result;
            } catch(e) {
                return _parse(text);
            }
        };
    """)
    
    page = context.new_page()
    page.goto("http://www.aqistudy.cn/historydata/monthdata.php?city=北京")
    
    for i in range(15):
        page.wait_for_timeout(2000)
        raw_data = page.evaluate("window.__intercepted_data__")
